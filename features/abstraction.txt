BEGIN PROGRAM

INITIALIZE SIMULATION:
    SET PLAYER_TIE_WIN = 0
    SET PLAYER_DRAGON_WIN = 0
    SET TOTAL_WIN_LOSS = 0

    DEFINE HOUSE_RULES:
        TIE_BET = TRUE, ODDS = 8:1
        DRAGON_BET = TRUE, ODDS = 40:1

    DEFINE SHUFFLES:
        SHOE = RANDOM
        RETURN SHUFFLED_SHOE

    DEFINE STRATEGY:
        PLAYER_BET = DEFAULT
        PLAYER_TIE_BET = DEFAULT
        PLAYER_DRAGON_BET = DEFAULT

        CONDITION:
            PLAYER_BET = DEFAULT_BET

            IF DECK CONTAINS 25% TENS THEN
                PLAYER_TIE_BET = ADJUSTED_VALUE
            ELSE IF DECK CONTAINS 40% SEVENS THEN
                PLAYER_DRAGON_BET = ADJUSTED_VALUE

            TOTAL_BET = PLAYER_BET + PLAYER_TIE_BET + PLAYER_DRAGON_BET
            RETURN TOTAL_BET

    DEFINE OUTCOME:
        COMBINE HOUSE_RULES, STRATEGY, AND SHUFFLES

RUN SIMULATION:
    CALCULATE RESULTS:
        PLAYER_PROFIT_LOSS = FUNCTION(OUTCOME)

    DISPLAY RESULTS

IF CONDITION (e.g., RE-RUN SIMULATION OR END) THEN
    RESTART PROGRAM
ELSE 
    END PROGRAM
